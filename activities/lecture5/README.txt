function getAnswer() {
    const question = document.getElementById("userQuestion").value.trim();
    const answerElement = document.getElementById("answer");

    if (question === "") {
        answerElement.textContent = "Please enter a question!";
        console.log("No question entered.");
        return;
    }

    const useMagic8Ball = Math.random() < 0.5;
    const chosenArray = useMagic8Ball ? magic8BallAnswers : fortuneCookieSayings;

    const randomIndex = Math.floor(Math.random() * chosenArray.length);
    const response = chosenArray[randomIndex];

    answerElement.textContent = `You asked: "${question}" \nAnswer: ${response}`;

    console.log(`Question: ${question}`);
    console.log(`Answer: ${response}`);

    return response;
}


The code above was created by adapting the function in lecture demo (L5V3).

How The function was implemented by extending the original demo code to support two arrays and DOM output.

Why The function was used because it handles fetching input, validation, random selection, and output display.

How The function was modified by adding logic to randomly select between magic8BallAnswers and fortuneCookieSayings, and by returning the chosen response.
